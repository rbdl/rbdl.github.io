610  bool validate_spatial_axis (Math::SpatialVector &axis) {
611 
612  bool axis_rotational = false;
613  bool axis_translational = false;
614 
615  Math::Vector3d rotation (axis[0], axis[1], axis[2]);
616  Math::Vector3d translation (axis[3], axis[4], axis[5]);
617 
618  if (fabs(rotation.norm()) > 1.0e-8)
619  axis_rotational = true;
620 
621  if (fabs(translation.norm()) > 1.0e-8)
622  axis_translational = true;
623 
624  if(axis_rotational && rotation.norm() - 1.0 > 1.0e-8) {
625  std::cerr << "Warning: joint rotation axis is not unit!" << std::endl;
626  }
627 
628  if(axis_translational && translation.norm() - 1.0 > 1.0e-8) {
629  std::cerr << "Warning: joint translation axis is not unit!" << std::endl;
630  }
631 
632  return axis_rotational != axis_translational;
633  }
