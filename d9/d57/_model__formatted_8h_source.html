401  Math::SpatialTransform GetJointFrame(unsigned int id) {
402  if (id >= fixed_body_discriminator) {
403  return mFixedBodies[id - fixed_body_discriminator].mParentTransform;
404  }
405 
406  unsigned int child_id = id;
407  unsigned int parent_id = lambda[id];
408  if (mBodies[parent_id].mIsVirtual) {
409  while (mBodies[parent_id].mIsVirtual) {
410  child_id = parent_id;
411  parent_id = lambda[child_id];
412  }
413  return X_T[child_id];
414  } else
415  return X_T[id];
416  }
