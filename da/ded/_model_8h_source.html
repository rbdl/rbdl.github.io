397  unsigned int GetParentBodyId (unsigned int id) {
398  if (id >= fixed_body_discriminator) {
399  return mFixedBodies[id - fixed_body_discriminator].mMovableParent;
400  }
401 
402  unsigned int parent_id = lambda[id];
403

414  Math::SpatialTransform GetJointFrame (unsigned int id) {
415  if (id >= fixed_body_discriminator) {
416  return mFixedBodies[id - fixed_body_discriminator].mParentTransform;
417  }
418 
419  unsigned int child_id = id;
420  unsigned int parent_id = lambda[id];
421  if (mBodies[parent_id].mIsVirtual) {
422  while (mBodies[parent_id].mIsVirtual) {
423  child_id = parent_id;
424  parent_id = lambda[child_id];
425  }
426  return X_T[child_id];
427  } else
428  return X_T[id];
429  }

434  void SetJointFrame (unsigned int id,
435  const Math::SpatialTransform &transform) {
436  if (id >= fixed_body_discriminator) {
437  std::cerr << "Error: setting of parent transform "
438  << "not supported for fixed bodies!" << std::endl;
439  abort();
440  }
441 
442  unsigned int child_id = id;
443  unsigned int parent_id = lambda[id];
444  if (mBodies[parent_id].mIsVirtual) {
445  while (mBodies[parent_id].mIsVirtual) {
446  child_id = parent_id;
447  parent_id = lambda[child_id];
448  }
449  X_T[child_id] = transform;
450  } else if (id > 0) {
451  X_T[id] = transform;
452  }
453  }
