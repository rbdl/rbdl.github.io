27 RBDL_DLLAPI bool LinSolveGaussElimPivot(MatrixNd A, VectorNd b, VectorNd &x) {
28  x = VectorNd::Zero(x.size());
29 
30  // We can only solve quadratic systems
31  assert(A.rows() == A.cols());
32 
33  unsigned int n = A.rows();
34  unsigned int pi;
35 
36  // the pivots
37  size_t *pivot = new size_t[n];
38 
39  // temporary result vector which contains the pivoted result
40  VectorNd px(x);
41 
42  unsigned int i, j, k;
43 
44  for (i = 0; i < n; i++) pivot[i] = i;
45 
46  for (j = 0; j < n; j++) {
47  pi = j;
48  double pv = fabs(A(j, pivot[j]));
49 
50  // LOG << "j = " << j << " pv = " << pv << std::endl;
51  // find the pivot
52  for (k = j; k < n; k++) {
53  double pt = fabs(A(j, pivot[k]));
54  if (pt > pv) {
55  pv = pt;
56  pi = k;
57  unsigned int p_swap = pivot[j];
58  pivot[j] = pivot[pi];
59  pivot[pi] = p_swap;
60  // LOG << "swap " << j << " with " << pi << std::endl;
61  // LOG << "j = " << j << " pv = " << pv << std::endl;
62  }
63  }
64 
65  for (i = j + 1; i < n; i++) {
66  if (fabs(A(j, pivot[j])) <= std::numeric_limits<double>::epsilon()) {
67  std::cerr << "Error: pivoting failed for matrix A = " << std::endl;
68  std::cerr << "A = " << std::endl << A << std::endl;
69  std::cerr << "b = " << b << std::endl;
70  }
71  // assert (fabs(A(j,pivot[j])) >
72  //std::numeric_limits<double>::epsilon());
73  double d = A(i, pivot[j]) / A(j, pivot[j]);
74 
75  b[i] -= b[j] * d;
76 
77  for (k = j; k < n; k++) {
78  A(i, pivot[k]) -= A(j, pivot[k]) * d;
79  }
80  }
81  }
82 
83  // warning: i is an unsigned int, therefore a for loop of the
84  // form "for (i = n - 1; i >= 0; i--)" might end up in getting an invalid
85  // value for i!
86  i = n;
87  do {
88  i--;
89 
90  for (j = i + 1; j < n; j++) {
91  px[i] += A(i, pivot[j]) * px[j];
92  }
93  px[i] = (b[i] - px[i]) / A(i, pivot[i]);
94 
95  } while (i > 0);
96 
97  // Unswapping
98  for (i = 0; i < n; i++) {
99  x[pivot[i]] = px[i];
100  }
101 
102  /*
103  LOG << "A = " << std::endl << A << std::endl;
104  LOG << "b = " << b << std::endl;
105  LOG << "x = " << x << std::endl;
106  LOG << "pivot = " << pivot[0] << " " << pivot[1] << " " << pivot[2] <<
107  std::endl; std::cout << LogOutput.str() << std::endl;
108  */
109 
110  delete[] pivot;
111 
112  return true;
113 }

362 RBDL_DLLAPI void SparseFactorizeLTL(Model &model, Math::MatrixNd &H) {
363  for (unsigned int i = 0; i < model.qdot_size; i++) {
364  for (unsigned int j = i + 1; j < model.qdot_size; j++) {
365  H(i, j) = 0.;
366  }
367  }
368 
369  for (unsigned int k = model.qdot_size; k > 0; k--) {
370  H(k - 1, k - 1) = sqrt(H(k - 1, k - 1));
371  unsigned int i = model.lambda_q[k];
372  while (i != 0) {
373  H(k - 1, i - 1) = H(k - 1, i - 1) / H(k - 1, k - 1);
374  i = model.lambda_q[i];
375  }
376 
377  i = model.lambda_q[k];
378  while (i != 0) {
379  unsigned int j = i;
380  while (j != 0) {
381  H(i - 1, j - 1) = H(i - 1, j - 1) - H(k - 1, i - 1) * H(k - 1, j - 1);
382  j = model.lambda_q[j];
383  }
384  i = model.lambda_q[i];
385  }
386  }
387 }

407 RBDL_DLLAPI void
408 SparseSolveLx(Model &model, Math::MatrixNd &L, Math::VectorNd &x) {
409  for (unsigned int i = 1; i <= model.qdot_size; i++) {
410  unsigned int j = model.lambda_q[i];
411  while (j != 0) {
412  x[i - 1] = x[i - 1] - L(i - 1, j - 1) * x[j - 1];
413  j = model.lambda_q[j];
414  }
415  x[i - 1] = x[i - 1] / L(i - 1, i - 1);
416  }
417 }
